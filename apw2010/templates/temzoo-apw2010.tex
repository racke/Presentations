\section{Templates}
Template::Zoom ist eine neue Templateengine, bei der im Gegensatz zu
Template::Toolkit das eigentliche Template aus reinem HTML besteht und somit
leicht von HTML-Designern bearbeitet werden kann.

Die Verknüpfung zu Objekten, Listen und Datenbankabfragen erfolgt über eine
Spezifikationsdatei.

\subsection{Ziele}
\begin{frame}{Warum Templates? Warum Template::Zoom?}
 \begin{itemize}
  \item Einheitliches Layout
  \item Trennung von Layout, Daten und Programmierung
  \item Einarbeitungszeit für Designer reduzieren
  \item Grenzfälle, Fehler und Bedingungen (dynamische Seiten)
 \end{itemize}
\end{frame}

\subsection{Bestandteile}
\begin{frame}{Bestandteile eines Template::Zoom-Template}
 \begin{itemize}
  \item Daten oder Objekte (Iterator)
  \item Spezifikation
  \item Template
 \end{itemize}
\end{frame}

\subsection{Warenkorb mit HTML::Zoom}
\begin{frame}[fragile]{Warenkorb: HTML::Zoom}
\begin{lstlisting}
use HTML::Zoom;

$cart = ...
$zoom = HTML::Zoom->from_file('cart.html');

$zoom->select('.cartitem')->repeat_content([
  map { my $field = $_; sub {
    $_[0]->select('.name')->replace_content($field->{title});
    $_[0]->select('.quantity')->replace_content($field->{quantity});
    $_[0]->select('.price')->replace_content($field->{price});
    };
  } @$cart]);

print $zoom->to_html();
\end{lstlisting}
\end{frame}

\section{Beispiel: Menüs}
\subsection{Datenbanktabelle für Menüs}
\begin{frame}[fragile]{Menüs: Datenbanktabelle}
\begin{lstlisting}
CREATE TABLE menus (
  code int NOT NULL auto_increment,
  name varchar(255) NOT NULL DEFAULT '',
  url  varchar(255) NOT NULL DEFAULT '',
  menu_name varchar(64) NOT NULL DEFAULT ''
  weight int NOT NULL DEFAULT 0,
  PRIMARY KEY(code),
  KEY(menu_name)
);
\end{lstlisting}
\end{frame}
